\newcommand{\centerimage}[4] {
	\begin{figure}[htb]
		\begin{center}
			\includegraphics[width=#4\linewidth]{#1}

		\end{center}
	\end{figure}
}

			\caption{#2}
			\label{fig:#3}
		\end{center}
	\end{figure}
}
